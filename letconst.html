LET w/ Looping

LET w/ Looping

  for(var i = 0; i < 10; i++){
    console.log(i); //Logs 0-9
  }
  for(let j = 0; j < 10; j++){
    console.log(j); //Logs 0-9
  }
  console.log(i); //Logs 10
  console.log(j); //ReferenceError
  • Loop vars aren't leaked anymore

LET w/ Looping


  for(let j = 0; j < 10; j++){
    let newNumber = j * 100;
  }
  console.log(newNumber); //ReferenceError
  • Loop bodies are a lexical scope

Block Functions

Block Functions

  function doThing(){
    //some code
    if(true){ //want to scope 'a', so if-true
      //...
      let a = 0;
      //...
    }
    //some code
  }
  • People would do this, just to get a temp scope

Block Functions

  function doThing(){
    //some code
    { //Lay down a new scope
      //...
      let a = 0;
      //...
    }
    //some code
  }
  • No if-true necessary
  • Same lexical scope as an if-true
